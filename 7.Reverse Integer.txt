class Solution {
    public int reverse(int x) {
        long ans = 0;
        while(x>0 || x < 0){
            int rem = x %10;                 
            ans = ans * 10 + rem;      
            x = x / 10;                
        }
        if(ans >= (long)Math.pow(2,31) -1 || ans <= (long)Math.pow(-2,31)) {
            return 0;
        }
        return (int)ans;       
    }
}
